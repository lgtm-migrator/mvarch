% Definition of log return
\begin{equationpage}{(1)}
  y_t = \mathop{\rm log} P_t - \mathop{\rm log} P_{t-1}
\end{equationpage}

%%%% UNIVARIATE CASE %%%%
% Univariate sampling
\begin{equationpage}{(2)}
  y_t = s_t z_t + \mu_t
\end{equationpage}

% Univariate mean update
\begin{equationpage}{(3)}
  \mu_t = a_u \mu_{t-1} + b_u y_{t-1} + c_u \bar y
\end{equationpage}% Univariate mean update

% Univariate ARMA model
\begin{equationpage}{(4)}
  y_t = (a_u + b_u) y_{t-1} + \varepsilon_t - a_u \varepsilon_{t-1}  + c_u \bar y
\end{equationpage}

% Univariate scale update
\begin{equationpage}{(5)}
  s_t^2 = (a_s s_{t-1})^2 + (b_s \varepsilon_{t-1})^2 + (c_s s)^2
\end{equationpage}

% Univariate matrix formulation
\begin{equationpage}{(6)}
  \begin{aligned}
    s_t^2 & = M_{t-1} M_{t-1}^T, \\
    \hbox{\rm where } \;
     M_t & = \left [
    \begin{matrix}
      a_s s_t &  b_s\varepsilon_t & c_s s
    \end{matrix}
    \right ]
  \end{aligned}
\end{equationpage}

%%%% MULTIVARIATE CASE %%%%
% Multivariate sampling
\begin{equationpage}{(7)}
  {\bf y}_t = S_t R_t {\bf z}_t + {\boldmu}_t
\end{equationpage}

% Multivariate mean update
\begin{equationpage}{(8)}
  {\boldmu}_t  = A_u {\boldmu}_{t-1} + B_u {\bf y}_{t-1} + C_u \bar {\bf y}
\end{equationpage}% Univariate mean update

% Multivariate scale update
\begin{equationpage}{(9)}
  \begin{aligned}
    S_t^2  & = M_{t-1} M_{t-1}^T, \\
    \hbox{\rm where } \;
    M_t & = \left [
      \begin{matrix}
        A_s S_t &  B_s \mathop{\rm diag}({\boldepsilon}_t) & C_s \mathop{\rm diag} ({\bf s})
      \end{matrix} \right ] \\
    \hbox{\rm and } \; {\bf s} & = \left [
      \begin{matrix}
        s_1, s_2, \ldots, s_n
      \end{matrix} \right ] \\
  \end{aligned}
\end{equationpage}

% Multivariate correlation update
\begin{equationpage}{(10)}
  \begin{aligned}
    R_t R_t^T  & = N_{t-1} N_{t-1}^T, \\
    \hbox{\rm where } \;
    N_t & = \left [
      \begin{matrix}
        A_r R_t &  B_rS_t^{-1} {\boldepsilon}_t & C_r R
      \end{matrix}
      \right ]
  \end{aligned}
\end{equationpage}
